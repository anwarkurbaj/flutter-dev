import 'dart:convert'; import 'package:http/http.dart' as http; class ProductRepository { Future<List<Product>> getProducts() async { // جيب الـ Token final token = await SharedPreferenceRepo.getToken(); // Headers final headers = { "Accept": "application/json", if (token != null) "Authorization": "Bearer $token", }; // الـ Request final response = await http.get( Uri.parse(ApiConstants.productsEndpoint), headers: headers, ); if (response.statusCode == 200) { final List data = jsonDecode(response.body)['data']; return data.map((item) => Product.fromJson(item)).toList(); } else { throw Exception('فشل في جلب البيانات'); } } }

📦 GetxController — القلب

كل منطق الصفحة يكون هون:

class HomeController extends GetxController { // ✅ المتغيرات القابلة للمراقبة (.obs) var counter = 0.obs; var name = ''.obs; var isLoading = false.obs; var products = <Product>[].obs; // ✅ يشتغل لما الـ Controller يتحمّل @override void onInit() { super.onInit(); fetchProducts(); } // ✅ يشتغل لما الـ Controller يتحذف @override void onClose() { super.onClose(); // تنظيف الـ controllers هون } void increment() => counter.value++; Future<void> fetchProducts() async { isLoading.value = true; try { final result = await _repo.getProducts(); products.assignAll(result); } catch (e) { print('Error: $e'); } finally { isLoading.value = false; } } }
👁️ Obx — مراقبة التغييرات في الـ UI

Obx بيراقب المتغيرات الـ .obs ويحدّث الـ UI تلقائياً:

// ✅ كل متغير .obs داخل Obx بيتحدث تلقائي Obx(() { if (controller.isLoading.value) { return CircularProgressIndicator(); } return Text(controller.name.value); }) // ✅ مثال مع ListView Obx(() => ListView.builder( itemCount: controller.products.length, itemBuilder: (ctx, i) => ProductCard(controller.products[i]), )) // ✅ مثال مع Counter Obx(() => Text( '${controller.counter.value}', style: TextStyle(fontSize: 24), ))
🔗 Get.put vs Get.find vs Get.lazyPut
// Get.put — ينشئ الـ Controller فوراً ويحفظه final controller = Get.put(HomeController()); // Get.find — يجيب الـ Controller الموجود مسبقاً final controller = Get.find<HomeController>(); // Get.lazyPut — ينشئ الـ Controller بس لما يُطلب Get.lazyPut(() => HomeController()); // في الـ StatelessWidget class HomeView extends StatelessWidget { final controller = Get.put(HomeController()); // ... } // في الـ StatefulWidget class _HomeState extends State<HomeView> { late HomeController controller; @override void initState() { super.initState(); controller = Get.put(HomeController()); } }

GetX Navigation
التنقل بين الصفحات بدون context
Get.to()
روح لصفحة جديدة (تضاف فوق الحالية)
Get.off()
روح لصفحة وأزل الحالية من الـ stack
Get.offAll()
روح لصفحة وامسح كل الـ stack
Get.back()
ارجع للصفحة السابقة
Get.toNamed()
روح لصفحة باستخدام اسم الـ Route
Get.offNamed()
روح لـ Route وأزل الحالية
Get.offAllNamed()
روح لـ Route وامسح كل الـ stack
Get.arguments
استقبل البيانات المرسلة من الصفحة السابقة

Shared Preferences
تخزين البيانات محلياً على الجهاز
🗄️ ما هو Shared Preferences؟

تخزين بيانات بسيطة (String, int, bool) على الجهاز — حتى بعد إغلاق التطبيق. مثل: التوكن، اسم المستخدم، الإعدادات.

💡

مناسب للبيانات الصغيرة. للبيانات الكبيرة استخدم SQLite أو Hive.

⚙️ SharedPreference Repository
import 'package:shared_preferences/shared_preferences.dart'; class SharedPreferenceRepo { // ✅ حفظ الـ Token static Future<void> saveToken(String token) async { final prefs = await SharedPreferences.getInstance(); await prefs.setString('token', token); } // ✅ جيب الـ Token static Future<String?> getToken() async { final prefs = await SharedPreferences.getInstance(); return prefs.getString('token'); } // ✅ حذف الـ Token (Logout) static Future<void> clearToken() async { final prefs = await SharedPreferences.getInstance(); await prefs.remove('token'); } // ✅ حفظ بيانات المستخدم كـ JSON static Future<void> saveUser(User user) async { final prefs = await SharedPreferences.getInstance(); await prefs.setString('user', jsonEncode(user.toJson())); } // ✅ جيب بيانات المستخدم static Future<User?> getUser() async { final prefs = await SharedPreferences.getInstance(); final str = prefs.getString('user'); if (str == null) return null; return User.fromJson(jsonDecode(str)); } // ✅ مسح كل شي (Logout) static Future<void> clearAll() async { final prefs = await SharedPreferences.getInstance(); await prefs.clear(); } }
